```
var Connection = new JsStore.Instance().openDb("Demo");
Connection.select({
    From: "Customers",
    Where:{
        country:'Mexico'
    },
    OnSuccess:function (results){
        log(results);
    },
    OnError:function (error) {
        log(error);
    }
});
```
